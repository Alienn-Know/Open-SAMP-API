109  // and scale up when rendering.
110  D3DCAPS9 d3dCaps;
111  m_pd3dDevice->GetDeviceCaps(&d3dCaps);
112 
113  if (m_dwTexWidth > d3dCaps.MaxTextureWidth)
114  {

120  hr = m_pd3dDevice->CreateTexture(m_dwTexWidth, m_dwTexHeight, 1,

162  SetTextAlign(hDC, TA_TOP);

165  // Meanwhile, keep track of the corresponding tex coords for each character.
166  DWORD x = 0;
167  DWORD y = 0;
168  TCHAR str[2] = _T("x");
169  SIZE size;
170

172  GetTextExtentPoint32(hDC, TEXT(" "), 1, &size);

179

181  {

183  y += size.cy + 1;

185

187

189  m_fTexCoords[c - 32][1] = ((FLOAT) (y + 0 + 0)) / m_dwTexHeight;
190  m_fTexCoords[c - 32][2] = ((FLOAT) (x + size.cx + m_dwSpacing)) / m_dwTexWidth;
191  m_fTexCoords[c - 32][3] = ((FLOAT) (y + size.cy + 0)) / m_dwTexHeight;
192 
193  x += size.cx + (2 * m_dwSpacing);
194  }
195 
196  // Lock the surface and write the alpha values for the set pixels
197  D3DLOCKED_RECT d3dlr;
198  m_pTexture->LockRect(0, &d3dlr, 0, 0);
199  BYTE* pDstRow = (BYTE*) d3dlr.pBits;
200  WORD* pDst16;

202 
203  for (y = 0; y < m_dwTexHeight; y++)
204  {
205  pDst16 = (WORD*) pDstRow;
206  for (x = 0; x < m_dwTexWidth; x++)
207  {
208  bAlpha = (BYTE) ((pBitmapBits[m_dwTexWidth*y + x] & 0xff) >> 4);
209  if (bAlpha > 0)
210  {
211  *pDst16++ = (WORD) ((bAlpha << 12) | 0x0fff);
212  }
213  else
214  {
215  *pDst16++ = 0x0000;
216  }
217  }
218  pDstRow += d3dlr.Pitch;
219  }
220 
221  // Done updating texture, so clean up used objects
222  m_pTexture->UnlockRect(0);
223  DeleteObject(hbmBitmap);
224  DeleteDC(hDC);
225  DeleteObject(hFont);
226 
227  return S_OK;
228 }
